def hitung_diskon(harga_awal, persen_diskon): """ Menghitung jumlah diskon dan harga akhir. :param harga_awal: float :param persen_diskon: float :return: tuple (jumlah_diskon, harga_akhir) """ jumlah_diskon = harga_awal * (persen_diskon / 100) harga_akhir = harga_awal - jumlah_diskon return jumlah_diskon, harga_akhir def main(): print("=== Aplikasi Penghitung Diskon ===") try: harga_awal = float(input("Masukkan harga awal: Rp ")) persen_diskon = float(input("Masukkan diskon (%): ")) if harga_awal < 0 or persen_diskon < 0: print("Error: harga dan diskon tidak boleh negatif.") return jumlah_diskon, harga_akhir = hitung_diskon(harga_awal, persen_diskon) print("\n=== Hasil Perhitungan ===") print(f"Jumlah Diskon : Rp {jumlah_diskon:,.2f}") print(f"Harga Akhir : Rp {harga_akhir:,.2f}") except ValueError: print("Input tidak valid!
